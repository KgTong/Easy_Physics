				$(options[i]).remove();

			$.each(data.channels, function(i, c) {
				var html = '';
				if (c.lvl == 1) {

				} else {
					html = "<option value='" + c.oid + "'>|-" + c.name
							+ "</option>";
				}
				$("#channelId").append(html);
			});
		});
	});

	function validatePwd() {
		var password = $("#password").val();
		var repassword = $("#repassword").val();
		if ($.trim(password) == "" || $.trim(repassword) == "") {
			alert("新密码或确认新密码不能为空！");
		} else {
			if ($.trim(password) != $.trim(repassword)) {
				alert("请保证两次输入的密码相同！");
			} else {
				document.forms[1].action = "resetPassWord.action";
				document.forms[1].submit();
			}
		}

	}
